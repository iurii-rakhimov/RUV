Word = ^Word
lenWord = LengthString(Word)
if lenWord == 1
{
	HideObject("Image")
	return
}
ShowObject("BackLetter")
ShowObject("SpaceBar")
SelectedIndex = GetIndexArrayForComponentWithNumber(^ArrayWord, Word, "/", 0)
^Index = SelectedIndex
//Print "SelectedIndex =" + SelectedIndex
//Print "Word =" + Word
NameImage = "Image_NO"
if SelectedIndex != -1
{
	if ^ArrayImage[SelectedIndex] == ""
	{
		if ^ArrayWord[SelectedIndex] == Word
		{
			NameImage = "Ok320x240"
		}
	}
	else
	{
		if ^ArrayWord[SelectedIndex] == Word
		{
//Print "SelectedIndex =" + SelectedIndex
			NameImage = ^ArrayImage[SelectedIndex]
//Print "NameImage =" + NameImage
		}
	}
}
Image.File = NameImage
//MiniImage.File = NameImage
HideObject("ListWordTable")
HideObject("Save")
if NameImage == "Ok320x240"
{
	Title.text = "Нет изображения, записать->"
	ShowObject("Edit")
}
else
{
	Title.text = ""
}
