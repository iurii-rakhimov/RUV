// QuessWord(KeyboardView).txt
if ^KeyboardView == "Show"
{
	for i = 1 ; i < 34 ;  i ++
	{
		if i < 10
		{
			HideObject("Let0"+i)
		}
		else
		{
			HideObject("Let"+i)
		}
	}
	^KeyboardView = "Hide"
}
else
{
	for i = 1 ; i < 34 ;  i ++
	{
		if i < 10
		{
			ShowObject("Let0"+i)
		}
		else
		{
			ShowObject("Let"+i)
		}
	}
	^KeyboardView = "Show"
}
