DoModule CleaWord
HideObject("Image")
HideObject("Save")
HideObject("Edit")
ShowObject("ListWordTable")
Title.text = ""
strSystemEvent = GetSystemEvent()
//print "strSystemEvent =" + strSystemEvent
if strSystemEvent != ""
{
	ArrayComponents = Components(strSystemEvent, " ")
	if ArrayComponents[0] == "SaveImageFile"
	{
		nameSaveImageFile = ArrayComponents[1]
		ArrayComponents = Components(nameSaveImageFile, "/")
		nameImageFile = ArrayComponents[1]
		print "nameImageFile =" + nameImageFile
		SelectedIndex = GetIndexArrayWithText(^ArrayImage, nameImageFile)
//		print "SelectedIndex =" + SelectedIndex
		^ArrayImage[SelectedIndex] = nameImageFile
		Title.text = "Запиши изменения ->"
		ShowObject("Save")
	}
}
