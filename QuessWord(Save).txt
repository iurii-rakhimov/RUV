// QuessWord(Save).txt

kv = 0
if kv == 0
{
	return
}
DelimiterNewLine = IntToChar(10)
CountRecords = CountElementsOfArray(^ArrayImage)
ContentFile = ""
for Index = 0 ; Index < CountRecords ;  Index ++
{
	Word = ^ArrayWord[Index]
	Image = ^ArrayImage[Index]
	if Image == ""
	{
		ContentFile = ContentFile + Word + DelimiterNewLine
	}
	else
	{
		ContentFile = ContentFile + Word + "*" + Image + DelimiterNewLine
	}
}
PathNameFile = ^Path + "QuessWord(AllWord).txt"
kv = FileWrite(PathNameFile, ContentFile)
if kv != 0
{
	strMessage="Ошибка записи в файл: " + PathNameFile
	Message(strMessage, "")
}
Title.text = "👍 Вывод ускорится,продолжайте"
HideObject("Save")
